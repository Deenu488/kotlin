assert(
    when {
        x == b -> true
        | |  |
        | |  3
        | false
        2

        x == a -> false
        | |  |
        | |  2
        | true
        2

        else -> true
    }
)

assert(
    when {
        a == A.ONE -> false
        | |    |
        | |    ONE
        | true
        ONE

        a == A.TWO -> true
        else -> true
    }
)
---
---
assert(
    when {
        a == A.ONE || a == A.TWO -> false
        | |    |
        | |    ONE
        | true
        ONE

        else -> true
    }
)

    when {
        x in 1..10 -> x == 3
        | |   |       | |
        | |   |       | false
        | |   |       2
        | |   1..10
        | true
        2

        else -> true
    }
)

    }
